-- Creating tables for Parkinsons ML model
CREATE TABLE parkinsons1 (	
	name VARCHAR(25) NOT NULL,	
	MDVP_Fo_Hz NUMERIC NOT NULL,
	MDVP_Fhi_Hz NUMERIC NOT NULL,
	MDVP_Flo_Hz NUMERIC NOT NULL,
	MDVP_Jitter_percent NUMERIC NOT NULL,
	MDVP_Jitter_Abs NUMERIC NOT NULL,
	MDVP_RAP NUMERIC NOT NULL,
	MDVP_PPQ NUMERIC NOT NULL,
	Jitter_DDP NUMERIC NOT NULL,
	MDVP_Shimmer NUMERIC NOT NULL,
	MDVP_Shimmer_db NUMERIC NOT NULL,
	PRIMARY KEY (name)	
);	

CREATE TABLE parkinsons2 (	
	name VARCHAR(25),
	Shimmer_APQ3 NUMERIC NOT NULL,
    	Shimmer_APQ5 NUMERIC NOT NULL,
    	MDVP_APQ NUMERIC NOT NULL,
    	Shimmer_DDA NUMERIC NOT NULL,
    	NHR NUMERIC NOT NULL,
    	HNR NUMERIC NOT NULL,
   	status NUMERIC NOT NULL,
    	RPDE NUMERIC NOT NULL,
    	DFA NUMERIC NOT NULL,
    	spread1 NUMERIC NOT NULL,
    	spread2 NUMERIC NOT NULL,
    	D2 NUMERIC NOT NULL,
    	PPE NUMERIC NOT NULL,
	FOREIGN KEY (name) REFERENCES parkinsons1 (name),	
	PRIMARY KEY (name)
);
